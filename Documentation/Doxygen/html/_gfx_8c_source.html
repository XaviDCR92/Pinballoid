330 static void GfxSortBigSprite(GsSprite* const psSpr)
331 {
332  enum
333  {
334  /* 64 = 2 ^ 5. */
335  GFX_TPAGE_WIDTH_BITSHIFT = 6
336  };
337 
338  /* On the other hand, GsSprite instances bigger than
339  * 256x256 px must be split into two primitives, so
340  * GsSortSprite shall be called twice. */
341 
342  /* Store original TPage, width and X data. */
343  const unsigned char aux_tpage = psSpr->tpage;
344  const short aux_w = psSpr->w;
345  const short aux_x = psSpr->x;
346 
347  /* First primitive will be 256x256 px. */
348  psSpr->w = MAX_SIZE_FOR_GSSPRITE;
349 
350  /* Render first primitive (256x256 px). */
351  GsSortSprite(psSpr);
352 
353  /* Second primitive will be:
354  * Width = Original Width - 256 px.
355  * Height = Original Height - 256 px. */
356  psSpr->x += MAX_SIZE_FOR_GSSPRITE;
357  psSpr->w = X_SCREEN_RESOLUTION - MAX_SIZE_FOR_GSSPRITE;
358 
359  /* TPage must be increased as we are looking 256 px to
360  * the right inside VRAM. Remember that TPages are 64x64 px. */
361  psSpr->tpage += MAX_SIZE_FOR_GSSPRITE >> GFX_TPAGE_WIDTH_BITSHIFT;
362 
363  /* Render second primitive. */
364  GsSortSprite(psSpr);
365 
366  /* Restore original TPage, width and X values. */
367  psSpr->tpage = aux_tpage;
368  psSpr->w = aux_w;
369  psSpr->x = aux_x;
370 }
